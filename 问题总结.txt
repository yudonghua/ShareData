  protected static String generateUrlWithToken(String url, String secretKey) {
        // token value = Md5(url + secretKey);
        url = formatUrl(url);
        String token = MD5Util.md5(url + secretKey);
        token = token + " ";
        if (url.indexOf("?") != -1) {
            return url + "&token=" + token;
        }

        return url + "?token=" + token;
    }
